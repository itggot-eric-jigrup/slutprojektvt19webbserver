post('/login') do
    result = login_user(params)
    name = params["name"]  
    if result[:error] == false
        session[:id] = result[:id]
        session[:name] = name
        redirect('/home')
    else 
        session[:error] = true
        session[:message] = result[:message] 
        redirect('/login')
    end
end
